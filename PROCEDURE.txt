  CREATE OR REPLACE PROCEDURE OT.INS_REGION(DATA2 VARCHAR2,ID OUT NUMBER)
    IS 
    V_ID NUMBER;
    BEGIN
    SELECT MAX(REGION_ID) INTO V_ID FROM OT.TEMP_REGION;
    INSERT INTO OT.TEMP_REGION(REGION_ID,REGION_NAME) VALUES (V_ID,DATA2);
    ID :=V_ID;
    END;
    /
    
    DECLARE
ID NUMBER;
DATA2 VARCHAR2(255) :='ASHWIN';
BEGIN
OT.INS_REGION(DATA2,ID);
dbms_output.put_line(ID);
END;
